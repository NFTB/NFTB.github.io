树状数组

1. 简单介绍

1.1 什么是树状数组

树状数组的本质思想是使用树结构维护“前缀和”，从而把时间复杂度降为$O(logn)$。

每个结点覆盖的长度为$lowbit(x)$
其中$lowbit(x)$是求出$x$在二进制数中最低位$1$及其后面的$0$构成的数
例如：$148=(100110100){_2}$最低位$1$和后面的$0$构成的数是$(100){_2}=4$
计算$lowbit(x)$:$lowbit(x)=x&(-x)$

1.2 树状数组的添加操作和查询操作

$add(x, k)$表示将序列中第$x$个数加上$k$。

从当前节点到根节点依次添加

void add(int x,int c){
    for(int i=x;i<=n;i+=lowbit(i)) tr[i]+=c;
}

$query(x)$表示将查询序列前$x$个数的和

从当前节点每次减$lowbit()$获取每段长为$lowbit()$和组成前缀和

int query(int x){
    int res=0;
    for(int i=x;i;i-=lowbit(i)) res+=tr[i];
    return res;
}

2. 应用

2.1 一维树状数组

前面所介绍的例子便是一维树状数组的单点修改区间查询操作
这里先介绍一下$O(n)初始化$
普通的初始化操作需要将$n$个数每个执行一次$add()$操作复杂度为$O(nlogn)$

展开代码 


for(int i=1;i<=n;i++) add(i,a[i]);//O(nlogn)初始化
for(int i=1;i<=n;i++){//O(n)初始化
    tr[i]+=a[i];
    int fa=i+lowbit(i);
    if(fa<=n){
        tr[fa]+=tr[i];
    }
}
for(int i=1;i<=n;i++) s[i]=s[i-1]+a[i];//O(n)初始化
for(int i=1;i<=n;i++) tr[i]=s[i]-s[i-lowbit(i)];

第一处$O(n)$初始化利用每个节点只更新一级父节点更新父节点时使用该节点的值来更新这便包括了子节点的所有子节点从而减少操作
第二处$O(n)$初始化利用的是前缀和原理，由于树状数组每个节点存储的的长度为$lowbit()$的前缀故可用这种方法直接给节点赋值

2.1.1 单点修改区间查询

例题：题目链接https://loj.ac/p/130/
前面介绍过这里直接给出代码

展开代码 


#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const int N=1e6+5;
int n,m;
ll tr[N],a[N];
int lowbit(int x){return (x&-x);}
void add(int x,int c){
    for(int i=x;i<=n;i+=lowbit(i)) tr[i]+=c;
}
ll query(int x){
    ll res=0;
    for(int i=x;i;i-=lowbit(i)) res+=tr[i];
    return res;
}
void slove(){
    cin>>n>>m;
    for(int i=1;i<=n;i++) cin>>a[i],add(i,a[i]);
    int op,l,r;
    while(m--){
        cin>>op>>l>>r;
        if(op==1) add(l,r);
        else cout<<query(r)-query(l-1)<<'\n';
    }
}
int main(){
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);std::cout.tie(nullptr);
    int _=1;
    // cin>>_;
    for(int i=1;i<=_;i++){
        slove();
    }
    return 0;
}

2.1.2 区间修改单点查询

例题：题目链接https://loj.ac/p/131/
树状数组为求前缀和的工具，那么将其用差分进行初始化把区间修改转换为差分两端点的修改，并通过求前缀和来得到单点的值。

展开代码 


#include<bits/stdc++.h>
using namespace std;

const int N=1e6+5;
int n,m;
typedef long long ll;
ll tr[N],a[N];
int lowbit(int x){return (x&-x);}
void add(int x,int c){
    for(int i=x;i<=n;i+=lowbit(i)) tr[i]+=c;
}
ll query(int x){
    ll res=0;
    for(int i=x;i;i-=lowbit(i)) res+=tr[i];
    return res;
}
void slove(){
    cin>>n>>m;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        tr[i]=a[i]-a[i-lowbit(i)];
    }
    int a,b,c,op;
    while(m--){
        cin>>op>>a;
        if(op==2) cout<<query(a)<<'\n';
        else{
            cin>>b>>c;
            add(a,c),add(b+1,-c);
        }
    }
}
int main(){
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);std::cout.tie(nullptr);
    int _=1;
    // cin>>_;
    for(int i=1;i<=_;i++){
        slove();
    }
    return 0;
}

2.1.3 区间修改区间查询

例题：https://loj.ac/p/132/
这里通过将差分初始化求出区间和的过程列出来
观察规律推出结果

通过推导可知完成区间修改区间查询需要两个树状数组共同完成

展开代码 


#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const int N=1e6+5;
int n,m;
ll tr1[N],tr2[N],a[N];
int lowbit(int x){return (x&-x);}
void add(ll x,ll c){
    for(ll i=x;i<=n;i+=lowbit(i)) tr1[i]+=c,tr2[i]+=x*c;
}
ll query(ll x){
    ll res=0;
    for(ll i=x;i;i-=lowbit(i)) res+=tr1[i]*(x+1)-tr2[i];
    return res;
}
void slove(){
    cin>>n>>m;
    for(int i=1;i<=n;i++) cin>>a[i];
    for(int i=1;i<=n;i++) add(i,a[i]-a[i-1]);
    ll l,r,d,op;
    while(m--){
        cin>>op;
        if(op==2){
            cin>>l>>r;
            cout<<query(r)-query(l-1)<<'\n';
        }else{
            cin>>l>>r>>d;
            add(l,d),add(r+1,-d);
        }
    }
}
int main(){
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);std::cout.tie(nullptr);
    int _=1;
    // cin>>_;
    for(int i=1;i<=_;i++){
        slove();
    }
    return 0;
}

2.2 二维树状数组

二维树状数组与一维差不多，注意在修改操作时是把该坐标往后的所有都进行修改，在查询操作时求出的是二维前缀和
了解过二维前缀和的话对这部分会好理解很多

2.2.1 单点修改区间查询

例题：https://loj.ac/p/133/
这里便是把修改和查询操作都改为了二维
在求和时利用二维前缀和的知识

通过下面的等式便很容易求出右下角矩形的和

展开代码 


#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const int N=5000;
ll tr[N][N];
int n,m;
int lowbit(int x){return (x&-x);}
void add(int x,int y,int c){
    for(int i=x;i<=n;i+=lowbit(i))
        for(int j=y;j<=m;j+=lowbit(j)) tr[i][j]+=c;
}
ll query(int x,int y){
    ll res=0;
    for(int i=x;i;i-=lowbit(i))
        for(int j=y;j;j-=lowbit(j)) res+=tr[i][j];
    return res;
}
void ans(int a,int b,int c,int d){
    cout<<query(c,d)-query(c,b-1)-query(a-1,d)+query(a-1,b-1)<<'\n';
}
void slove(){
    cin>>n>>m;
    int op,x,y,k,d;
    while(cin>>op){
        if(op==1) {
            cin>>x>>y>>k;
            add(x,y,k);
        }else{
            cin>>x>>y>>k>>d;
            ans(x,y,k,d);
        }
    }
}
int main(){
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);std::cout.tie(nullptr);
    int _=1;
    // cin>>_;
    for(int i=1;i<=_;i++){
        slove();
    }
    return 0;
}

2.2.2 区间修改单点查询

例题：https://loj.ac/p/134
二维差分的操作在这里不在过多赘述
代码中的修改操作利用的便是二维差分

展开代码 


#include<bits/stdc++.h>
using namespace std;

const int N=5000;
typedef long long ll;
ll tr[N][N];
int n,m;
int lowbit(int x){return (x&-x);}
void add(int x,int y,int c){
    for(int i=x;i<=n;i+=lowbit(i))
        for(int j=y;j<=m;j+=lowbit(j))
            tr[i][j]+=c;
}
void fun(int a,int b,int c,int d,int k){
    add(a,b,k);add(c+1,d+1,k);
    add(a,d+1,-k);add(c+1,b,-k);
}
ll query(int x,int y){
    ll res=0;
    for(int i=x;i;i-=lowbit(i))
        for(int j=y;j;j-=lowbit(j))
            res+=tr[i][j];
    return res;
}
void slove(){
    cin>>n>>m;
    int op,a,b,c,d,k;
    while(cin>>op){
        if(op==1){
            cin>>a>>b>>c>>d>>k;
            fun(a,b,c,d,k);
        }else{
            cin>>a>>b;
            cout<<query(a,b)<<'\n';
        }
    }
}
int main(){
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);std::cout.tie(nullptr);
    int _=1;
    // cin>>_;
    for(int i=1;i<=_;i++){
        slove();
    }
    return 0;
}

2.2.3 区间修改区间查询

例题：https://loj.ac/p/135
大致思路和一维差不多将需要求的区间和列出来进行填补可能需要一点三维想象这里我给出CSDN上一篇博客列出的推导公式

$$\sum_{i=1}^x\sum_{j=1}^y\sum_{k=1}^i\sum_{h=1}^j d[i][j]*(x+1-i) * (y+1-j)$$

展开得到
$$(x+1)*(y+1)*\sum_{i=1}^x\sum_{j=1}^yd[i][j]$$
$$-(y+1)*\sum_{i=1}^x\sum_{j=1}^yd[i][j]*i$$
$$-(x+1)*\sum_{i=1}^x\sum_{j=1}^yd[i][j]*j$$
$$+\sum_{i=1}^x\sum_{j=1}^yd[i][j] * i * j$$

这四个式子便可用四个树状数组维护

展开代码 


#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
const int N=2050;
ll tr1[N][N],tr2[N][N],tr3[N][N],tr4[N][N];
int n,m;
int lowbit(int x){return (x&-x);}
void add(int x,int y,ll c){
    for(int i=x;i<=n;i+=lowbit(i))
        for(int j=y;j<=m;j+=lowbit(j)){
            tr1[i][j]+=c;tr2[i][j]+=x*c;
            tr3[i][j]+=y*c;tr4[i][j]+=x*y*c;
        }
}
void real_add(int a,int b,int c,int d,int x){
    add(a,b,x);add(c+1,d+1,x);add(a,d+1,-x);add(c+1,b,-x);
}
ll query(int x,int y){
    ll res=0;
    for(int i=x;i;i-=lowbit(i))
        for(int j=y;j;j-=lowbit(j))
            res+=tr1[i][j]*(x+1)*(y+1)-(y+1)*tr2[i][j]-(x+1)*tr3[i][j]+tr4[i][j];
    return res;
}
ll real_query(int a,int b,int c,int d){
    return query(c,d)+query(a-1,b-1)-query(a-1,d)-query(c,b-1);
}
void slove(){
    cin>>n>>m;
    int op,a,b,c,d,x;
    while(cin>>op){
        if(op==1){
            cin>>a>>b>>c>>d>>x;
            real_add(a,b,c,d,x);
        }else{
            cin>>a>>b>>c>>d;
            cout<<real_query(a,b,c,d)<<'\n';
        }
    }
}
int main(){
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);std::cout.tie(nullptr);
    int _=1;
    // cin>>_;
    for(int i=1;i<=_;i++){
        slove();
    }
    return 0;
}

2.3求逆序对

例题：https://www.luogu.com.cn/problem/P1908
经典的求逆序对是用归并排序，这里介绍一下用树状数组来求
读入数据时纪录一下每个点的位置，进行排序。
从后往前依次把，每个点的位置添加到树状数组中
每次通过求前缀和便可知道排序后在该数后面但位置却在该数前面的有多少个，求出总和便逆序对数

展开代码 


#include<bits/stdc++.h>
using namespace std;

const int N=5e5+5;
int n;
int tr[N];
struct node{
    int x,pos;
};
int lowbit(int x){return (x&-x);}
void add(int x){
    for(int i=x;i<=n;i+=lowbit(i)) tr[i]++;
}
int query(int x){
    int res=0;
    for(int i=x;i;i-=lowbit(i)) res+=tr[i];
    return res;
}
void slove(){
    cin>>n;
    vector<node>v;
    for(int i=0;i<n;i++){
        int k;
        cin>>k;
        v.push_back({k,i+1});
    }
    sort(v.begin(),v.end(),[&](node a,node b){
        if(a.x==b.x) return a.pos<b.pos;
        return a.x<b.x;
    });
    long long ans=0;
    for(int i=n;i>=1;i--){
        ans+=query(v[i-1].pos-1);
        add(v[i-1].pos);
    }
    cout<<ans<<'\n';
}
int main(){
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);std::cout.tie(nullptr);
    int _=1;
    // cin>>_;
    for(int i=1;i<=_;i++){
        slove();
    }
    return 0;
}

3.总结

树状数组是一个非常强大的工具，虽然功能不如线段树，但其代码量少，在解决特定的一类问题时非常方便
